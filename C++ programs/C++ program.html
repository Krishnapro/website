#include <iostream>

int money=100;

class player{
	char name[30];
	int age;
	public:
		void set_player(char *nm,int ag){
			strcpy(name,nm);
			age = ag;
		}
		void get_player(){
			cout<<"player name :- "<<name<<endl;
			cout<<"age :- "<<age<<endl;
			cout<<"\t\t\t\tCASH remaining :-"<<money<<endl;
		}
};

void head(){

void card(){

	cout<<"    |___|        |___|        |___|  \n";

void start(){
	char name [30];
	int age;
	
	head();
	cout<<"\t\t\t\tEnter Player Details"<<endl<<"Enter Player Name :- ";

	head();
	player player1;

	player1.get_player();
}

int play(){
	char ch[3] ={'j','k','q'};
	int x,y,temp,guess;
	head();
	
	cout<<"guess king is at which position...\n"<<endl;
	card();
	cout<<"shuffling...\n";
	cin>>guess;
	
	for(int i;i<5;i++)
	{
		x = rand() % 3;
		y = rand() % 3;
		temp = ch[x];
		ch[x] = ch[y];
		ch[y] = temp;
	}

	if(ch[guess-1]=='k')
		return 1;
	else
		return 0;
}
int main() 
{
	int bet,result;
	start();
	while(money>0)
	{	head();

		cin>>bet;
		if(bet == 0|| bet>money)
			break;
		result=play();
		if(result == 0)
		{	cout<<"you lose\n";
			money = money-bet;

		{
			cout<<"you won\n";
			money=money+bet*2;
		}
		cout<<"your balance = "<<money<<endl;
	}

}
